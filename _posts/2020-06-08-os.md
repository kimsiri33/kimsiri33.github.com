```python
import os
while 1:
    a=os.getcwd()
    b=input(a+'>')
    if b[0:3]=='pwd':
        print(a)
    elif b[0:3]=='dir':
        b = os.popen("dir")
        print(b.read())
    elif b=='cd ..':
        c=a.split('\\')
        f=c[:-1]
        d='\\'.join(f)
        os.chdir(d)
    elif b[0:2]=='cd':
        print(os.chdir(b[3:]))
    elif b[0:3]=='cat':
        d=open(a+'\\'+b[4:], 'r')
        while True:
            g = d.readline()
            if not g: break
            print(g)
```
